Mouse Actions

Recommended: Please solve it by yourself, before moving on to the solution.

    import java.applet.Applet;
    import java.awt.Color;
    import java.awt.Font;
    import java.awt.Label;
    import java.awt.event.MouseEvent;
    import java.awt.event.MouseListener;
    public class Mouse extends Applet{
        public void init()
        { 
            setLayout(null);
            Label l=new Label("Hello");
            Label l2=new Label("Location:");
            l.setFont(new Font("Serif",Font.BOLD,30));
            l.setBounds(30, 30, 300, 50);
            l2.setBounds(30, 80, 100, 20);
            add(l);
            add(l2);
            
                class hand implements MouseListener{
                @Override
                public void mouseClicked(MouseEvent e) {
                    l2.setText("Location: "+e.getX()+","+e.getY());
                    // TODO Auto-generated method stub
                    
                }
    
                @Override
                public void mousePressed(MouseEvent e) {
                    // TODO Auto-generated method stub
                    l.setBackground(Color.blue);
                    
                }
    
                @Override
                public void mouseReleased(MouseEvent e) {
                    l.setFont(new Font("Serif",Font.BOLD,26));
                    // TODO Auto-generated method stub
                    
                }
    
                @Override
                public void mouseEntered(MouseEvent e) {
                    // TODO Auto-generated method stub
                    l.setText("Swing");
    
                }
    
                @Override
                public void mouseExited(MouseEvent e) {
                    // TODO Auto-generated method stub
                    l.setText("Hello");
                    l.setBackground(getBackground());
                    
                }                
            }
            l.addMouseListener(new hand());
        }
    }
