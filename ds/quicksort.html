Quick Sort

            #include<stdlib.h>
            int partition(int a[],int beg,int end)
            {
                int rear=end,front=beg,i=beg,temp;
                int piv=a[beg];
                step1:{
                    while(front<=rear)
                    {
                        if(a[rear]<piv)
                        {
                            temp=a[rear];
                            a[rear]=a[i];
                            a[i]=temp;
                            i=rear;
                            goto step2;
                        }
                        rear--;
                    }
                }
                step2:{
                    while(front<=rear)
                    {
                        if(a[front]>piv)
                        {
                            temp=a[front];
                            a[front]=a[i];
                            a[i]=temp;
                            i=front;
                            goto step1;
                        }
                        front++;
                    }
                
                }
                return i;
            }
            void quickSort(int a[],int beg,int end)
            {
                int index;
                if(beg<end)
                {
                    index=partition(a,beg,end);
                    quickSort(a,beg,index-1);
                    quickSort(a,index+1,end);
                }
            }
