Double Ended Queue

Recommended: Please solve it by yourself, before moving on to the solution.

    #include<stdio.h>
    #include<conio.h>
    #include<stdlib.h>
    #define max 7
    int deque[max];
    int front=-1,rear=-1;
    int isEmpty()
    {
        if(front==-1)
        return 1;
        else return 0;
    }
    int isFull()
    {
        if(front==0&&rear==max-1||front==rear+1)
        return 1;
        else
        return 0;
    }
    void InsertFE(int item)
    {
        if(isFull())
        {
            printf("Queue Overflow!");return;
        }	
            if(isEmpty())
            {
                front=rear=0;	
            }
            else if(front==0)
            front=max-1;
            else
            front-=1;
            deque[front]=item;
    }
    void InsertRE(int item)
    {
        if(isFull())
        {
            printf("Queue Overflow!");return;
        }	
            if(isEmpty())
            {
                front=rear=0;	
            }
            else if(rear==max-1)
            rear=0;
            else
            rear+=1;
            deque[rear]=item;
    }
    int delFE()
    {
        if(isEmpty())
        {
                printf("Queue underflow!");
                exit(0);
        }
        int item=deque[front];
        if(front==rear)
        front=rear=-1;
        else if(front==max-1)
        front=0;
        else
            front+=1;
        return item;
    }
    int delRE()
    {
        if(isEmpty())
        {
                printf("Queue underflow!");
                exit(0);
        }
        int item=deque[rear];
        if(front==rear)
        front=rear=-1;
        else if(rear==0)
        rear=max-1;
        else
            rear-=1;
        return item;
    }
    void display()
    {
        int i;
        if(isEmpty())
        {
        printf("Queue is Empty.");
        return;
        }
        i=front;		
            if(front<=rear)
            while(i<=rear)
            printf("%d ",deque[i++]);
            else
            {
                while(i<=max-1)
                printf("%d ",deque[i++]);
                i=0;
                while(i<=rear)
                printf("%d ",deque[i++]);
            }
    }
    int main()
    {
        int n,c;
        while(1)
        {
            system("cls");
            printf("1.Insert at Front End\n2.Insert at Rear End\n3.Delete at Front End
            \n4.Delete at Rear End\n5.Display\n6.Exit\nEnter your choice: ");
            scanf("%d",&c);
            int s;
            switch(c)
            {
                case 1:printf("Enter a number: ");
                    scanf("%d",&n);
                    InsertFE(n);break;
                case 2:printf("Enter a number: ");
                    scanf("%d",&n);
                    InsertRE(n);break;       
                case 3:printf("%d",delFE());break;
                case 4:printf("%d",delRE());break;
                case 5:display();break;
                case 6:exit(0);
                default:printf("wrong choice!");
            }
            getch();
        }
        return 0;
    }
