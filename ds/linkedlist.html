    struct node{
        int info;
        struct node *link;
    };
    struct node *start=NULL;
    struct node *head=NULL;
    void insertNode(int item)
    {
        struct node *temp,*t;
        temp=(struct node *)malloc(sizeof(struct node));
        temp->info=item;
        temp->link=NULL;
        if(start==NULL)
        {
            start=temp;
        } 
        else
        {
            t=start;
            while(t->link!=NULL)
            t=t->link;
            t->link=temp;
        }
    }
