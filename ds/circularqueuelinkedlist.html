Circular Queue Using Linked List

Recommended: Please solve it by yourself, before moving on to the solution.

            #include<stdio.h>
            #include<conio.h>
            #include<stdlib.h>
            struct node
            {
                int data;
                struct node *link;
            };
            struct node *head=NULL;
            int isEmpty()
            {
                if(head==NULL)
                return 1;
                else 
                return 0;
            }
            void insert(int item)
            {
                struct node *p;
                p=(struct node *)malloc(sizeof(struct node));
                if(p==NULL)
                {
                    printf("Memory not available.");
                    return;
                }
                p->data=item;
                if(isEmpty())
                {
                head=p;
                head->link=head;
                }
                else
                {
                    p->link=head->link;
                    head->link=p;
                    head=p;
                }
            }
            int del()
            {
                struct node *temp;
                if(isEmpty())
                {
                    printf("Underflow");
                    exit(1);
                }
                else if(head->link==head)
                {
                    temp=head;
                    head=NULL;
                }
                else
                {
                    temp=head->link;
                    head->link=temp->link;//head=link=head->link->link;
                }
                int item=temp->data;
                free(temp);
                return item;
            }
            void display()
            {
                struct node *t;
                if(isEmpty())
                {
                    printf("Queue is Empty.");
                    return;
                }
                t=head->link;
                do{
                    printf("%d ",t->data);
                    t=t->link;
                }while(t!=head->link);
            }
            int main()
            {
            int n,c;
            while(1)
            {
                system("cls");
                printf("1.Insert\n2.Delete\n3.Display\n4.Exit
                \nEnter your choice: ");
                scanf("%d",&c);
                switch(c)
                {
                    case 1:printf("Enter a number: ");
                            scanf("%d",&n);
                            insert(n);break;
                    case 2:printf("Deleted item= %d",del());break;
                    case 3:display();break;
                    case 4:exit(0);
                    
                }
                getch();
            }
            return 0;
            }
