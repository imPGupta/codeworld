Queue Using Dynamic Array

Recommended: Please solve it by yourself, before moving on to the solution.

    #include<stdio.h>
    #include<conio.h>
    #include<stdlib.h>
    #include<malloc.h>
    struct queue
    {
        int front,rear,capacity;
        int *array;
    };
    struct queue* createQueue(int cap)
    {
        struct queue *q;
        q=(struct queue*)malloc(sizeof(struct queue));
        q->front=-1;
        q->rear=-1;
        q->capacity=cap;
        q->array=(int*)malloc(sizeof(int)*q->capacity);
        if(!q->array) return NULL;
        return q;	
    }
    int isFullQueue(struct queue *q)
    {
        if((q->front==0&&q->rear==q->capacity-1)||q->front==q->rear+1)
        return 1;
        else return 0;
    }
    void Insert(struct queue *q,int data)
    {
        if(q->front==-1)
        {
                q->front=q->rear=0;
                q->array[q->rear]=data;	
        }
        else
        {
            if(!isFullQueue(q))
            {
                if(q->rear==q->capacity-1)
                {
                    q->rear=0;
                    q->array[q->rear]=data;
                }
            
                else
                {
                q->rear+=1;
                q->array[q->rear]=data;
                }
            }
            else
            printf("Queue is Full, element not inserted.");
        }
    }
    void Delete(struct queue *q)
    {
        if(q->front==-1)
        printf("Queue is empty.");
        else if(q->front==q->rear)
        {
            printf("Deleted item= %d",q->array[q->front]);
            q->front=q->rear=-1;
        }
        else if(q->front==q->capacity-1)
        {
            printf("Deleted item= %d",q->array[q->front]);
            q->front=0;
        }
        else
        {
            printf("Deleted item= %d",q->array[q->front]);
            q->front+=1;
        }
    }
    void View(struct queue *q)
    {
        int t;
        t=q->front;
        if(t==-1)
        printf("Queue is Empty.");
        else
        {
            while(t!=q->rear)
            {
                printf("%d ",q->array[t]);
                if(t==q->capacity-1)
                t=0;
                else
                t++;
            }
            printf("%d ",q->array[t]);
        }
    }    
    int main()
    {
    int n,c;
    struct queue *q=createQueue(5);
    while(1)
    {
        system("cls");
        printf("1.Insert\n2.Delete\n3.View\n4.Exit\nEnter your choice: ");
        scanf("%d",&c);
        switch(c)
        {
            case 1:printf("Enter a number: ");
                    scanf("%d",&n);
                    Insert(q,n);break;
            case 2:Delete(q);break;
            case 3:View(q);break;
            case 4:exit(0);
        }
        getch();
    }
    return 0;
    }
