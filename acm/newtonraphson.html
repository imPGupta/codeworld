Newton Raphson

Recommended: Please solve it by yourself, before moving on to the solution.

    #include<stdio.h>
    #include<conio.h>
    #include<math.h>
    double fun(double x)
    {
        return (x*x*x-5*x+1);
    }    
    double fund(double x)
    {
        return (3*x*x-5);
    }
    void nr(double x)
    {
        double h=(fun(x)/fund(x));
        while(fabs(h)>=0.000001)
        {   
            x=x-h;
            h=(fun(x)/fund(x));
        
        }
        printf("\nroot is %lf",x); 
    }
    int main()
    {
        double x;
        printf("Enter the point ");
        scanf("%lf",&x);
        nr(x);   
    }
